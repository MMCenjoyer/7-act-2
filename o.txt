using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace ConsoleApp23
{
    internal class Program
    {
        static void Main(string[] args)
        {
            int opcion;
            do
            {
                Console.WriteLine("\n Menu Principal");
                Console.WriteLine("\n 1.Calcula suma");
                Console.WriteLine("\n 2.Calcula resta");
                Console.WriteLine("\n 3.Salir");
                Console.WriteLine("\n Selecciones una opci√≥n");
                opcion = Convert.ToInt32(Console.ReadLine());

                {
                    Console.WriteLine(" Error opcion no valida. Por favor ingrese 1,2,3");
                }
                else if (opcion == 1 || opcion == 2)
                {
                    Console.WriteLine("ingrese el primer numero");

                    Console.WriteLine("ingrese el segundo numero");
                    double num2 = Convert.ToDouble(Console.ReadLine());
                    double resultado = 0;
                    string operador = "";
                    if (opcion == 1)
                    {
                        resultado = num1 + num2;
                        operador = "+";
                        Console.WriteLine($"\n Suma: {num1} + {num2} = {resultado}");

                }
